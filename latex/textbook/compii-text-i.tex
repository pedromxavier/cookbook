	\section*{Herança}

	\begin{problem}{O Bidicionário}
	
	\proposal
	Construir uma classe \code{Bidict}, que herde as propriedades de \type{dict}, mas que garanta a paridade entre chaves e valores. Isto é, se num bidicionário \code{b}, temos que se \code{b[x] = y} então \code{b[y] = x}.

	\end{problem}
